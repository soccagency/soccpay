🚀 SoccPay SDK

Ejemplo completo de integración en navegador

💳 Demo de Pago

⚠️ Modo Demo Este ejemplo usa credenciales de prueba. Para pagos reales, configura tus credenciales de SoccPay.

📋 Código de Integración

🌐 Usando CDN (Más Fácil)

<!-- Cargar SoccPay SDK desde CDN -->
<script src="https://unpkg.com/soccpay@latest/dist/index.min.js"></script>

<script>
// El SDK está disponible globalmente
const soccPay = new SoccPay.SoccPay();

// Crear pago
const payment = soccPay.createPayment()
    .setCredentials({
        client_id: 'tu_client_id',
        client_secret: 'tu_client_secret'
    })
    .setPayer(
        soccPay.createPayer().setPaymentMethod('SoccPay')
    )
    .setTransaction(
        soccPay.createTransaction()
            .setAmount(
                soccPay.createAmount()
                    .setTotal(49.99)
                    .setCurrency('USD')
            )
    )
    .setRedirectUrls(
        soccPay.createRedirectUrls()
            .setSuccessUrl(window.location.origin + '/success')
            .setCancelUrl(window.location.origin + '/cancel')
    );

// Procesar pago
async function procesarPago() {
    try {
        const checkoutUrl = await payment.create();
        window.location.href = checkoutUrl;
    } catch (error) {
        console.error('Error:', error);
    }
}
</script>
